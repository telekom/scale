import ScaleCallout from './ScaleCallout.vue';

<Meta
  title="Beta Components/Callout"
  component={ScaleCallout}
  argTypes={{
    variant: {
      defaultValue: 'primary',
      options: ['primary', 'white', 'black', 'blue'],
      control: {
        type: 'select',
      },
    },
    size: {

      control: {

      },
    },

      },
    },

        type: 'text',
      },
    },
  }}

export const Template = (args, { argTypes }) => ({
  props: { ...ScaleCallout.props,
    prefix: String,
    text: String
    },
  template: `
    <scale-callout :size="size" :variant="variant" :rotation="rotation" :asterisk="asterisk">
      <div slot="prefix">{{ prefix }}</div>
      <span slot="text">{{ text }}</span>
    </scale-callout>
  `,
});

### Scoped CSS variables
```css
 :host {
  --color-callout: var(--scl-color-primary);
  --inner-width-small: var(--scl-spacing-80);
  --inner-height-small: var(--scl-spacing-80);
  --inner-width-large: 126.5px;
  --inner-height-large: 96px;
  --width-small: 120px;
  --height-small: 120px;
  --height-large: 160px;
  --width-large: 160px;
  --blue-callout: var(--scl-color-blue-60);
  --white-callout: var(--scl-color-white);
  --black-callout: var(--scl-color-black);
  --primary-callout: var(--scl-color-primary);
}
}
```

## Size
There are two sizes: 
Large / small. Large has the size 160x160 and small 120x120. In the example below, the callout is small
