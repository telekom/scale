                              >

                                    padding: '1px 3px 1px 3px',

                                  }}
                                  onClick={() => {
                                    navigator.clipboard.writeText(
                                      getJsPathNameByPathString(pathString)

                                  }}
                                >
                                  <scale-icon-action-copy-paste

                                    accessibility-title="copy-paste"
