export const TemplateInteractive = (args, { argTypes }) => ({
  components: { ScaleChip },
  template: `
  <div>
    <div style="height: 20px;">Dismisable options:</div>
      <ul aria-label="selection" style="display: flex; list-style: none; padding: 0;">
        <li v-for="chipLabel in labels" :key="chipLabel" style="margin-right: 4px;">
          <ScaleChip
            type="dynamic"
            :label="chipLabel"
            :selected="selectedChip === chipLabel"
            @scale-close="removeChip(chipLabel)"
          >
            {{chipLabel}}
          </ScaleChip>
        </li>
       
      </ul>   
    </div>
  `,
  methods: {
    removeChip(label) {
      
      action('scale-close')(label);
      this.labels = this.labels.filter(l => l !== label);
      this.selectedChip = this.labels[0];
      this.$nextTick(() => {
        const firstChip = this.$el.querySelector('ul > li > scale-chip');
        if (firstChip && firstChip.shadowRoot) {
          const button = firstChip.shadowRoot.querySelector('span > button');
          if (button) button.focus();
        }
      });
    },
  },
  data: () => ({
    labels: ['option 1', 'option 2', 'option 3', 'option 4'],
    selectedChip: 'option 1'
  }),
});
