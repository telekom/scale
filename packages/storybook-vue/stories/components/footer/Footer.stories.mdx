import { footerNavigation } from '../brand-header-navigation/fixtures';
import AppFooter from './AppFooter.vue';
const setColorMode = () => {
  try {
    document.body.dataset.mode = JSON.parse(
      window.localStorage.getItem('persistedColorMode')
    );
  } catch (e) {}
};
